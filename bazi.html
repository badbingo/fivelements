            function displayBasicInfo(info) {

                const lines = info.split('\n').filter(line => line.trim());

               

                // 提取八字四柱

                const pillarsLine = lines.find(line => line.includes('八字四柱'));

                if (pillarsLine) {

                    const pillars = pillarsLine.match(/年柱\[(.*?)\] 月柱\[(.*?)\] 日柱\[(.*?)\] 时柱\[(.*?)\]/);

                    if (pillars && pillars.length === 5) {

                        yearPillar.textContent = pillars[1];

                        monthPillar.textContent = pillars[2];

                        dayPillar.textContent = pillars[3];

                        hourPillar.textContent = pillars[4];

                    }

                }

               

                // 提取本命局十神

                const tenGodsLine = lines.find(line => line.includes('本命局十神'));

                if (tenGodsLine) {

                    const tenGodsMatch = tenGodsLine.match(/本命局十神：\[(.*?)\]/);

                    if (tenGodsMatch && tenGodsMatch[1]) {

                        const allGods = tenGodsMatch[1].split(',');

                       

                        // Clear previous content

                        stemGodsList.innerHTML = '';

                        branchGodsList.innerHTML = '';

                       

                        // First 4 are stem gods (天干)

                        allGods.slice(0, 4).forEach((god, index) => {

                            if (god.trim()) {

                                const item = document.createElement('div');

                                item.className = 'ten-gods-item';

                                item.textContent = `${['年干','月干','日干','时干'][index]}: ${god.trim()}`;

                                stemGodsList.appendChild(item);

                            }

                        });

                       

                        // Next 4 are branch gods (地支)

                        allGods.slice(4, 8).forEach((god, index) => {

                            if (god.trim()) {

                                const item = document.createElement('div');

                                item.className = 'ten-gods-item';

                                item.textContent = `${['年支','月支','日支','时支'][index]}: ${god.trim()}`;

                                branchGodsList.appendChild(item);

                            }

                        });

                    }

                }

               

                // 提取藏干情况

                const hiddenLine = lines.find(line => line.includes('藏干情况'));

                if (hiddenLine) {

                    const hidden = hiddenLine.match(/年柱\[(.*?)\] 月柱\[(.*?)\] 日柱\[(.*?)\] 时柱\[(.*?)\]/);

                    if (hidden && hidden.length === 5) {

                        yearHidden.textContent = hidden[1] || '-';

                        monthHidden.textContent = hidden[2] || '-';

                        dayHidden.textContent = hidden[3] || '-';

                        hourHidden.textContent = hidden[4] || '-';

                    }

                }

               

                // 提取命格等级

                const fateLine = lines.find(line => line.includes('命格等级'));

                if (fateLine) {

                    const fateMatch = fateLine.match(/命格等级：\[(.*?)\]/);

                    if (fateMatch && fateMatch[1]) {

                }

            function getStarRating(ratingText) {

                // 将数字评分转换为星标

                const rating = parseInt(ratingText);

                if (isNaN(rating)) return '-';

               

                const fullStars = '★'.repeat(rating);

                const emptyStars = '☆'.repeat(5 - rating);

                return fullStars + emptyStars;

            }

           

            function displaySectionContent(section, result, contentElement) {

                // 处理评分显示

                if (result.includes('★')) {

                    result = result.replace(/(★+)/g, '<span class="rating">$1</span>');

                    result = result.replace(/(☆+)/g, '<span style="color:#ddd">$1</span>');

                }

               

                // 去除特殊符号

                result = result.replace(/[#*\、]/g, '');

               

                // 处理命运总结的特殊格式

                if (section === 'summary') {

                    // 提取主要优缺点和人生建议

                    const strengthsWeaknessesMatch = result.match(/3\s*主要优缺点：([\s\S]*?)4\s*人生建议：/);

                    const adviceMatch = result.match(/4\s*人生建议：([\s\S]*)/);

                   

                    let formattedResult = '';

                   

                    if (strengthsWeaknessesMatch && strengthsWeaknessesMatch[1]) {

                        formattedResult += `<div class="section-title">主要优缺点</div>`;

                        formattedResult += `<p>${strengthsWeaknessesMatch[1].trim()}</p>`;

                    }

                   

                    if (adviceMatch && adviceMatch[1]) {

                        formattedResult += `<div class="section-title">人生建议</div>`;

                        formattedResult += `<p>${adviceMatch[1].trim()}</p>`;

                    }

                   

                    // 添加剩余内容

                    const remainingContent = result.replace(/3\s*主要优缺点：[\s\S]*/, '');

                    if (remainingContent.trim()) {

                        formattedResult = `<p>${remainingContent.trim()}</p>` + formattedResult;

                    }

                   

                    contentElement.innerHTML = formattedResult;

                    return;

                }

               

                // 处理其他部分的标题和内容样式

                result = result.replace(/([^\n]+：)/g, '<div class="section-title">$1</div>');

               

                // 处理换行

                result = result.split('\n').map(line => {

                    if (line.trim() === '') return '';

                    if (line.startsWith('<div class="section-title">')) return line;

                    return `<p>${line}</p>`;

                }).join('');

               

                contentElement.innerHTML = result;

            }

        });
