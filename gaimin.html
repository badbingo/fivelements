            function getStarsDisplay(stars) {

                let html = '<div class="stars">';

                for (let i = 0; i < 5; i++) {

                    if (i < stars) {

                        // 根据星星数量选择不同的图标

                        if (stars === 1) {

                            html += '<i class="fas fa-bomb bomb-icon"></i>';

                        } else if (stars === 2) {

                            html += '<i class="fas fa-exclamation-triangle warning-icon"></i>';

                        } else if (stars === 3) {

                            html += '<i class="fas fa-balance-scale balance-icon"></i>';

                        } else if (stars === 4) {

                            html += '<i class="fas fa-star star-icon"></i>';

                        } else if (stars === 5) {

                            html += '<i class="fas fa-thumbs-up thumb-icon"></i>';

                        }

                    } else {

                        html += '<i class="far fa-star star-empty"></i>';

                    }

                }

                html += '</div>';

                return html;

            }

            function displaySectionContent(section, result, contentElement) {

                // 处理评分显示

                if (result.includes('★')) {

                    result = result.replace(/(★+)/g, '<span style="color:#FFD700">$1</span>');

                    result = result.replace(/(☆+)/g, '<span style="color:#ddd">$1</span>');

                }

               

                // 处理1-5星评级为图标

                result = result.replace(/1星/g, '<i class="fas fa-bomb bomb-icon"></i>');

                result = result.replace(/2星/g, '<i class="fas fa-exclamation-triangle warning-icon"></i>');

                result = result.replace(/3星/g, '<i class="fas fa-balance-scale balance-icon"></i>');

                result = result.replace(/4星/g, '<i class="fas fa-star star-icon"></i>');

                result = result.replace(/5星/g, '<i class="fas fa-thumbs-up thumb-icon"></i>');

               

                // 去除特殊符号

                result = result.replace(/[#*\、]/g, '');

               

                // 处理标题和内容样式

                result = result.replace(/([^\n]+：)/g, '<div class="section-title">$1</div>');

               

                // 创建卡片容器

                const cardHtml = `<div class="fortune-card">${result}</div>`;

               

                // 处理换行

                const formattedResult = cardHtml.split('\n').map(line => {

                    if (line.trim() === '') return '';

                    if (line.startsWith('<div class="section-title">')) return line;

                    return `<p>${line}</p>`;

                }).join('');

               

                contentElement.innerHTML = formattedResult;

            }

        });
